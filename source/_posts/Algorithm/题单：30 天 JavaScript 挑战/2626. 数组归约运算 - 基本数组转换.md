# 2626. 数组归约运算

> Problem: [2626. 数组归约运算](https://leetcode.cn/problems/array-reduce-transformation/description/)

```JS
/**
 * @param {number[]} nums
 * @param {Function} fn
 * @param {number} init
 * @return {number}
 */
var reduce = function(nums, fn, init) {
    var val=init
    for(i in nums){
        val=fn(val,nums[i])
    }
    return val
};
```

```JS
/**
 * @param {number[]} nums
 * @param {Function} fn
 * @param {number} init
 * @return {number}
 */
var reduce = function(nums, fn, init) {
    return nums.reduce((foldVar,value,index)=>{
      return fn(foldVar,value)

};
```
