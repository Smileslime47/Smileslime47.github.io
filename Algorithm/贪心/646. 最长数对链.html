646. жңҖй•ҝж•°еҜ№й“ҫ #

еҺҹйўҳең°еқҖпјҡhttps://leetcode.cn/problems/maximum-length-of-pair-chain/

йўҳи§Ј #

ж–№жі•дёҖпјҡеҠЁжҖҒи§„еҲ’

е°ҶpairsиҝӣиЎҢжҺ’еәҸпјҢз„¶еҗҺз”ЁдёҖдёӘж•°з»„dpи®°еҪ•д»Ҙpairs[i]з»“е°ҫзҡ„жңҖй•ҝж•°й“ҫ

д»ӨjйҒҚеҺҶ0еҲ°iпјҢеҪ“j[1]<i[0]ж—¶пјҢиҜҙжҳҺiеҸҜд»ҘжҺҘеңЁд»Ҙjз»“е°ҫзҡ„жңҖй•ҝж•°й“ҫеҗҺеҪўжҲҗдёҖдёӘжӣҙй•ҝзҡ„ж•°й“ҫ

жӯӨж—¶dp[i]=dp[j]+1пјҢи®°еҪ•жңҖеӨ§зҡ„dp[i]еҚідёәзӯ”жЎҲ

ж—¶й—ҙеӨҚжқӮеәҰпјҡO(N^2)

з©әй—ҙеӨҚжқӮеәҰпјҡO(N)

kotlin
class Solution {
    fun findLongestChain(pairs: Array<IntArray>): Int {
        pairs.sortBy { a -> a[0] }
        val dp = Array(pairs.size) { 1 }
        var max = 1
        for (i in 0 until pairs.size) {
            for (j in 0 until i) {
                if (pairs[j][1] < pairs[i][0]) {
                    dp[i] = if (dp[i] < dp[j] + 1) dp[j] + 1 else dp[i]
                    max = if (max < dp[i]) dp[i] else max
                }
            }
        }
        return max
    }
}

ж–№жі•дәҢпјҡиҙӘеҝғ

е’Ңдјҡи®®е®үжҺ’зұ»дјјпјҢе°ҶpairsжҢүз…§еҸіиҫ№з•ҢиҝӣиЎҢеҚҮеәҸжҺ’еәҸ

еҪ“жҜҸж¬ЎйғҪеңЁж»Ўи¶іжқЎд»¶зҡ„ж•°еҜ№дёӯйҖүеҸіиҫ№з•ҢжңҖе°Ҹзҡ„ж•°еҜ№жһ„жҲҗж•°й“ҫж—¶пјҢжңҖеҗҺзҡ„з»“жһңиҮӘз„¶жҳҜжңҖдјҳзҡ„

ж—¶й—ҙеӨҚжқӮеәҰпјҡO(N)

з©әй—ҙеӨҚжқӮеәҰпјҡO(1)

kotlin
class Solution {
    fun findLongestChain(pairs: Array<IntArray>): Int {
        pairs.sortBy { a -> a[1] }
        var max = 0
        var rightWall=Int.MIN_VALUE
        for (i in 0 until pairs.size) {
            if(pairs[i][0]>rightWall){
                rightWall=pairs[i][1]
                max++
            }
        }
        return max
    }
}
