class Solution {
    public int numDecodings(String s) {
        long[][] dp=new long[s.length()][2];
        if(s.charAt(0)=='*'){
            dp[0][0]=9;
        }else if(s.charAt(0)=='0'){
            dp[0][0]=0;
        }else{
            dp[0][0]=1;
        }

        for(int i=1;i<s.length();i++){
            if(s.charAt(i)=='*'){
                dp[i][0]=(dp[i-1][0]+dp[i-1][1])*9;
            }else if(s.charAt(i)=='0'){
                dp[i][0]=0;
            }else{
                dp[i][0]=(dp[i-1][0]+dp[i-1][1])*1;
            }

            if(s.charAt(i-1)=='1'){
                if(s.charAt(i)=='*'){
                    dp[i][1]=dp[i-1][0]*9;
                }else{
                    dp[i][1]=dp[i-1][0]*1;
                }
            }else if(s.charAt(i-1)=='2'){
                if(s.charAt(i)=='*'){
                    dp[i][1]=dp[i-1][0]*6;
                }else if(s.charAt(i)<='6'){
                    dp[i][1]=dp[i-1][0]*1;
                }
            }else if(s.charAt(i-1)=='*'){
                if(s.charAt(i)=='*'){
                    dp[i][1]=(i>1?dp[i-2][0]+dp[i-2][1]:1)*15;
                }else if(s.charAt(i)<='6'){
                    dp[i][1]=(i>1?dp[i-2][0]+dp[i-2][1]:1)*2;
                }else{
                    dp[i][1]=(i>1?dp[i-2][0]+dp[i-2][1]:1)*1;
                }
            }

            dp[i][0]%=1000000007;
            dp[i][1]%=1000000007;
        }
        return (int)(dp[s.length()-1][0]+dp[s.length()-1][1])%1000000007;
    }
}
