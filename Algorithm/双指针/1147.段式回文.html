class Solution {
    public int longestDecomposition(String text) {
        int leftEdge=0,rightEdge=text.length()-1;
        int i,j,k,ans=0;
        boolean isEqual;
        while(leftEdge<=rightEdge){
            isEqual=false;
            i=leftEdge;
            j=rightEdge;

            while(!isEqual){
                for(k=0;j+k<=rightEdge;k++){
                    if(text.charAt(leftEdge+k)!=text.charAt(j+k)){
                        isEqual=false;
                        break;
                    }
                    isEqual=true;
                }

                if(isEqual==false){
                    i++;
                    j--;
                }else{
                    if(i==rightEdge||j==leftEdge){
                        leftEdge=999;
                        rightEdge=0;
                        ans+=1;
                    }else{
                        leftEdge=i+1;
                        rightEdge=j-1;
                        ans+=2;
                    }
                }
            }
        }
        return ans;
    }
}
