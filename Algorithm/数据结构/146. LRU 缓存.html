class LRUCache {
    class ListNode{
        int key;
        int val;
        ListNode next;
        ListNode last;
        ListNode(int k,int v){
            this.key=k;
            this.val=v;
        }
    }
    //ه“ˆه¸Œç»‘ه®ڑKeyه’Œه¯¹ه؛”èٹ‚ç‚¹
    Map<Integer,ListNode> idx=new HashMap<>();
    //è™ڑو‹ںه¤´èٹ‚ç‚¹ï¼Œvalè®°ه½•é“¾è،¨é•؟ه؛¦
    ListNode head=new ListNode(0,0);
    ListNode tail=head;
    int capacity;
    void removeNode(ListNode node){
        if(node==tail){
            tail=node.last;
            tail.next=null;
        }else{
            node.last.next=node.next;
            node.next.last=node.last;
        }
        head.val--;
    }
    void addNode(ListNode node){
        tail.next=node;
        node.last=tail;
        tail=tail.next;
        tail.next=null;
        head.val++;
    }
    void refreshKey(int key){
        removeNode(idx.get(key));
        addNode(idx.get(key));
    }

    public LRUCache(int capacity) {
        this.capacity=capacity;
    }

    //ه¦‚و‍œو‰¾هˆ°هˆ™refreshه…³é”®ه­—ï¼Œç„¶هگژè؟”ه›‍valï¼Œو‰¾ن¸چهˆ°هˆ™è؟”ه›‍-1
    public int get(int key) {
        if(idx.containsKey(key)){
            refreshKey(key);
            return idx.get(key).val;
        }
        else return -1;
    }

    public void put(int key, int value) {
        //ه¦‚و‍œه­کهœ¨èٹ‚ç‚¹هˆ™و›´و–°ه€¼ه¹¶refreshه…³é”®ه­—
        if(idx.containsKey(key)){

                removeNode(head.next);
            }
            //و·»هٹ و–°èٹ‚ç‚¹
            addNode(new ListNode(key,value));
            idx.put(key,tail);
        }
    }
