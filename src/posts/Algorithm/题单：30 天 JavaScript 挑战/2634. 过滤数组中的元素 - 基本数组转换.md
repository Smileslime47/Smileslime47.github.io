> Problem: [2634. 过滤数组中的元素](https://leetcode.cn/problems/filter-elements-from-array/description/)上

```JS
/**
 * @param {number[]} arr
 * @param {Function} fn
 * @return {number[]}
 */
var filter = function(arr, fn) {
    const array=[]
    arr.forEach((value,index)=>{
        if(fn(value,index)){
            array.push(value)
        }
    })
    return array
};
```

```JS
/**
 * @param {number[]} arr
 * @param {Function} fn
 * @return {number[]}
 */
var filter = function(arr, fn) {
    const array = []

    for(let i=0;i<arr.length;i++){
        if(fn(arr[i],i)){
            array.push(arr[i])
        }
    }

    return array
}
```

```JS
/**
 * @param {number[]} arr
 * @param {Function} fn
 * @return {number[]}
 */
var filter = function(arr, fn) {
    var array = []

    array = arr.reduce((reduceArr, value, index) => {
        if (fn(value, index)) reduceArr.push(value)
        return reduceArr
    }, [])

    return array
}
```

```JS
/**
 * @param {number[]} arr
 * @param {Function} fn
 * @return {number[]}
 */
var filter = function(arr, fn) {
    var array = []
    array = arr.filter(fn)

    return array
}
```
