class Solution {
    public int integerBreak(int n) {
        int[] dp=new int[n+1];
        boolean[] available=new boolean[n+1];
        dp[1]=1;
        available[1]=true;

        int mul;
        boolean flag;
        for(int i=2;i<=n;i++){
            mul=0;
            flag=false;
            for(int j=1;j<n;j++){
                if(i-j>=0&&available[i-j]){
                    mul=Math.max(mul,Math.max(dp[i-j],i-j)*Math.max(dp[j],j));
                    flag=true;
                }
            }
            if(flag){
                dp[i]=mul;
                available[i]=true;
            }
        }

        if(available[n])
            return dp[n];
        else
            return -1;
    }
}
